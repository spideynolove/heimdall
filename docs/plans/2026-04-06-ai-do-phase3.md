**Goal:** Build `ai-do` — a keyword-based task router that reads `models.yaml` + `routing.yaml` from `~/.ai-profiles/orchestrator/`, selects the best model, and dispatches the task via `ai_dispatch.dispatch()`. Supports `--dry-run` for routing preview and `--model` to force a specific model.

**Architecture:** Single stdlib-only script `src/ai_do.py`. Reads YAML via `re`+stdlib (no PyYAML dependency — parse the simple subset we need), or accepts PyYAML if already in the venv. Routing is pure keyword matching: tokenise the task string, check each rule's `match` list, first-match wins. Falls back to `claude` if no rule matches.

**Tech Stack:** Python 3.8+ (stdlib + optional PyYAML), pytest + `unittest.mock`, same `tmp_path` fixture pattern as previous phases.

```python
#!/usr/bin/env python3
from __future__ import annotations
import sys
from pathlib import Path


def load_yaml(path: Path) -> dict:
    if not path.exists():
        return {}
    try:
        import yaml
        return yaml.safe_load(path.read_text()) or {}
    except ImportError:
        pass
    return _parse_yaml_simple(path.read_text())


def _parse_yaml_simple(text: str) -> dict:
    import re
    result: dict = {}
    stack: list = [result]
    indent_stack: list[int] = [-1]
    current_list: list | None = None
    list_key: str = ""
    list_parent: dict | None = None

    for raw_line in text.splitlines():
        if not raw_line.strip() or raw_line.strip().startswith("#"):
            continue
        indent = len(raw_line) - len(raw_line.lstrip())
        line = raw_line.strip()

        # list item
        if line.startswith("- "):
            content = line[2:].strip()
            if current_list is not None:
                # inline list item that is a scalar
                current_list.append(content)
            continue

        # inline list: key: [a, b, c]
        inline = re.match(r'^(\w[\w-]*):\s*\[(.+)\]$', line)
        if inline:
            key = inline.group(1)
            vals = [v.strip() for v in inline.group(2).split(",")]
            while indent <= indent_stack[-1]:
                stack.pop()
                indent_stack.pop()
            current_container = stack[-1]
            if isinstance(current_container, dict):
                current_container[key] = vals
            current_list = None
            continue

        # key: scalar
        kv = re.match(r'^(\w[\w-]*):\s*(.+)$', line)
        if kv:
            key, val = kv.group(1), kv.group(2).strip()
            while indent <= indent_stack[-1]:
                stack.pop()
                indent_stack.pop()
            current_container = stack[-1]
            if isinstance(current_container, dict):
                current_container[key] = val
            current_list = None
            continue

        # key: (mapping start)
        km = re.match(r'^(\w[\w-]*):$', line)
        if km:
            key = km.group(1)
            while indent <= indent_stack[-1]:
                stack.pop()
                indent_stack.pop()
            current_container = stack[-1]
            if isinstance(current_container, list):
                new: dict = {}
                current_container.append(new)
                stack.append(new)
            elif isinstance(current_container, dict):
                if key not in current_container:
                    current_container[key] = {}
                stack.append(current_container[key])
            indent_stack.append(indent)
            current_list = None
            continue

    return result
```

> **Note:** The stdlib YAML parser is intentionally minimal — it handles only the schema in `models.yaml` and `routing.yaml`. If it proves brittle, add PyYAML to dev deps: `uv pip install pyyaml` and `[project.optional-dependencies] dev = [..., "pyyaml"]`. The `load_yaml` function will prefer PyYAML automatically.

```python
import re as _re


def _tokenise(text: str) -> set[str]:
    return set(_re.split(r'[^a-z0-9_]+', text.lower()))


def route(task: str, rules: list[dict]) -> tuple[str, str]:
    tokens = _tokenise(task)
    for rule in rules:
        if tokens & set(rule.get("match", [])):
            return rule.get("primary", "claude"), rule.get("fallback", "backup")
    return "claude", "backup"
```

```python
import os
import sys


def dispatch_task(
    task: str,
    models_path: Path | None = None,
    routing_path: Path | None = None,
    force_model: str = "",
    dry_run: bool = False,
    session: str = "ai-do",
) -> str:
    orc_dir = Path(os.environ.get("AI_PROFILES_DIR", Path.home() / ".ai-profiles")) / "orchestrator"
    if models_path is None:
        models_path = orc_dir / "models.yaml"
    if routing_path is None:
        routing_path = orc_dir / "routing.yaml"

    data_models = load_yaml(models_path).get("models", {})
    data_routing = load_yaml(routing_path).get("rules", [])

    if force_model:
        primary = force_model
        fallback = (data_models.get(primary) or {}).get("failover", "backup")
    else:
        primary, fallback = route(task, data_routing)

    if dry_run:
        print(f"[dry-run] task: {task!r}")
        print(f"[dry-run] primary: {primary}  fallback: {fallback}")
        cmd = get_ccs_cmd(primary, data_models)
        print(f"[dry-run] ccs command: {cmd}")
        return ""

    profiles_dir = Path(os.environ.get("AI_PROFILES_DIR", Path.home() / ".ai-profiles"))
    profile = primary if (profiles_dir / primary).is_dir() else ""

    import ai_dispatch
    return ai_dispatch.dispatch(task, profile=profile, session=session)
```

```python
def main() -> None:
    args = sys.argv[1:]
    if not args or args[0] != "run":
        print("usage: ai-do run <task> [--model <name>] [--dry-run] [--session <name>]", file=sys.stderr)
        sys.exit(1)
    args = args[1:]
    task = ""
    force_model = ""
    dry_run = False
    session = os.environ.get("AI_DISPATCH_SESSION", "ai-do")
    i = 0
    while i < len(args):
        if args[i] == "--model" and i + 1 < len(args):
            force_model = args[i + 1]
            i += 2
        elif args[i] == "--session" and i + 1 < len(args):
            session = args[i + 1]
            i += 2
        elif args[i] == "--dry-run":
            dry_run = True
            i += 1
        elif not task:
            task = args[i]
            i += 1
        else:
            i += 1
    if not task:
        print("error: task string required", file=sys.stderr)
        sys.exit(1)
    result = dispatch_task(task, force_model=force_model, dry_run=dry_run, session=session)
    if result:
        print(result)


if __name__ == "__main__":
    main()
```
